def fibonacci(n):
    """Generate fibonacci sequence up to n terms."""
    a, b = 0, 1
    for _ in range(n):
        print(a, end=' ')
        a, b = b, a + b
    print()

# Usage

FROM users u

WHERE u.active = true

ORDER BY post_count DESC
LIMIT 10;
